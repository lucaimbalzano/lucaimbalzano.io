## Usage

```tsx title='app/layout.tsx'
import { Toaster } from '@simbashrd/ui'

const Layout = ({ children }) => {
  return (
    <html>
      <body>
        {children}
        <Toaster />
      </body>
    </html>
  )
}

export default Layout
```
